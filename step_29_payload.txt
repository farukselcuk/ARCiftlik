```javascript
import * as THREE from 'three';

export class DayNightCycle {
  constructor(scene, renderer) {
    this.scene = scene;
    this.renderer = renderer;

    // ── GÜNEŞ ──
    this.sun = this.createSun();
    scene.add(this.sun);

    // ── AY ──
    this.moon = this.createMoon();
    scene.add(this.moon);

    // ── YILDIZLAR ──
    this.stars = this.createStars();
    scene.add(this.stars);

    // ── IŞIK KAYNAKLARI ──
    this.sunLight = new THREE.DirectionalLight(0xffffff, 1.0);
    this.sunLight.castShadow = true;
    scene.add(this.sunLight);

    this.moonLight = new THREE.DirectionalLight(0x4466aa, 0.2);
    scene.add(this.moonLight);

    this.ambientLight = new THREE.AmbientLight(0x404040, 0.5);
    scene.add(this.ambientLight);
  }

  createSun() {
    const group = new THREE.Group();

    // Güneş küresi
    const sun = new THREE.Mesh(
      new THREE.SphereGeometry(0.18, 16, 12),
      new THREE.MeshBasicMaterial({ color: 0xFFDD44 })
    );
    group.add(sun);

    // Güneş ışınları (hale)
    for (let i = 0; i < 8; i++) {
      const ray = new THREE.Mesh(
        new THREE.BoxGeometry(0.04, 0.22, 0.01),
        new THREE.MeshBasicMaterial({
          color: 0xFFEE66, transparent: true, opacity: 0.4
        })
      );
      ray.rotation.z = (i / 8) * Math.PI * 2;
      ray.position.z = -0.01;
      group.add(ray);
    }

    return group;
  }

  createMoon() {
    const moon = new THREE.Mesh(
      new THREE.SphereGeometry(0.12, 14, 10),
      new THREE.MeshStandardMaterial({
        color: 0xDDDDCC,
        roughness: 0.9,
        emissive: 0x888877,
        emissiveIntensity: 0.15
      })
    );
    // Ay kraterleri — küçük çukurlar
    moon.userData.isMoon = true;
    return moon;
  }

  createStars() {
    const count = 200;
    const geometry = new THREE.BufferGeometry();
    const positions = new Float32Array(count * 3);
    for (let i = 0; i < count; i++) {
      const theta = Math.random() * Math.PI * 2;
      const phi   = Math.random() * Math.PI;
      const r     = 8 + Math.random() * 2;
      positions[i * 3]     = r * Math.sin(phi) * Math.cos(theta);
      positions[i * 3 + 1] = Math.abs(r * Math.cos(phi)); // Sadece üstte
      positions[i * 3 + 2] = r * Math.sin(phi) * Math.sin(theta);
    }
    geometry.setAttribute('position', new THREE.BufferAttribute(positions, 3));
    return new THREE.Points(
      geometry,
      new THREE.PointsMaterial({ color: 0xffffff, size: 0.02, transparent: true, opacity: 0 })
    );
  }

  // Gerçek saate göre döngü — 0.0 (gece yarısı) → 0.5 (öğlen) → 1.0 (gece yarısı)
  getDayProgress() {
    const now = new Date();
    return (now.getHours() * 3600 + now.getMinutes() * 60 + now.getSeconds()) / 86400;
  }

  // Her frame çağır
  update() {
    const t = this.getDayProgress(); // 0..1
    const angle = t * Math.PI * 2 - Math.PI / 2; // Sabah doğudan başlar

    const radius = 3.5;

    // Güneş pozisyonu (yay üzerinde)
    this.sun.position.set(
      Math.cos(angle) * radius,
      Math.sin(angle) * radius,
      -1.5
    );

    // Ay güneşin karşısında
    this.moon.position.set(
      Math.cos(angle + Math.PI) * radius,
      Math.sin(angle + Math.PI) * radius,
      -1.5
    );

    // Güneş ufkun altındaysa gizle
    const sunVisible = this.sun.position.y > -0.3;
    this.sun.visible = sunVisible;

    // Ay gece görünür
    const moonVisible = !sunVisible || this.moon.position.y > 0.5;
    this.moon.visible = moonVisible;

    // Yıldızlar sadece geceleyin
    const nightness = Math.max(0, -this.sun.position.y / radius);
    this.stars.material.opacity = Math.min(nightness * 1.5, 0.9);

    // Işık şiddeti — güneş pozisyonuna göre
    const sunHeight = Math.max(0, this.sun.position.y / radius);
    this.sunLight.intensity = sunHeight * 1.2;
    this.sunLight.position.copy(this.sun.position);

    // Ay ışığı
    this.moonLight.intensity = nightness * 0.3;
    this.moonLight.position.copy(this.moon.position);

    // Ortam ışığı — gündüz açık, gece koyu
    this.ambientLight.intensity = 0.2 + sunHeight * 0.6;

    // Gökyüzü rengi — sabah pembesi, öğlen mavi, akşam turuncusu, gece lacivert
    this.updateSkyColor(t);

    // Güneş ışınları animasyonu
    this.sun.rotation.z += 0.002;
  }

  updateSkyColor(t) {
    // t: 0=gece yarısı, 0.25=sabah, 0.5=öğlen, 0.75=akşam
    let skyColor, fogColor;

    if (t < 0.2 || t > 0.85) {
      // Gece
      skyColor = new THREE.Color(0x0a0a1a);
      fogColor = new THREE.Color(0x0a0a1a);
    } else if (t < 0.28) {
      // Şafak — mor/pembe
      skyColor = new THREE.Color(0xff7043).lerp(new THREE.Color(0x87ceeb), (t - 0.2) / 0.08);
      fogColor = skyColor;
    } else if (t < 0.72) {
      // Gündüz — açık mavi
      skyColor = new THREE.Color(0x87ceeb);
      fogColor = new THREE.Color(0xb0d4e8);
    } else {
      // Akşam — turuncu/kırmızı
      skyColor = new THREE.Color(0x87ceeb).lerp(new THREE.Color(0xff5722), (t - 0.72) / 0.13);
      fogColor = skyColor;
    }

    this.renderer.setClearColor(skyColor);
    if (this.scene.fog) this.scene.fog.color.copy(fogColor);
  }

  dispose() {
    this.scene.remove(this.sun, this.moon, this.stars);
    this.scene.remove(this.sunLight, this.moonLight, this.ambientLight);
  }
}
```

```javascript
export class LeafSystem {
  constructor(scene) {
    this.scene = scene;
    this.leaves = [];
    this.active = false;
  }

  start() {
    if (this.active) return;
    this.active = true;
    // 30 yaprak
    for (let i = 0; i < 30; i++) {
      const leaf = new THREE.Mesh(
        new THREE.PlaneGeometry(0.04, 0.05),
        new THREE.MeshBasicMaterial({
          color: [0xFF8C42, 0xFFB347, 0xCC5200, 0xFFD700][Math.floor(Math.random()*4)],
          side: THREE.DoubleSide, transparent: true, opacity: 0.85
        })
      );
      leaf.position.set(
        (Math.random() - 0.5) * 5,
        1 + Math.random() * 2,
        (Math.random() - 0.5) * 5
      );
      leaf.userData = {
        vx: (Math.random() - 0.5) * 0.3,
        vy: -(0.1 + Math.random() * 0.2),
        vz: (Math.random() - 0.5) * 0.3,
        rotSpeed: (Math.random() - 0.5) * 0.05
      };
      this.scene.add(leaf);
      this.leaves.push(leaf);
    }
  }

  update(delta) {
    if (!this.active) return;
    this.leaves.forEach(leaf => {
      const d = leaf.userData;
      leaf.position.x += d.vx * delta;
      leaf.position.y += d.vy * delta;
      leaf.position.z += d.vz * delta;
      leaf.rotation.z += d.rotSpeed;
      // Yere düştüyse yukarıdan tekrar başlat
      if (leaf.position.y < -0.2) {
        leaf.position.set(
          (Math.random() - 0.5) * 5,
          1.5 + Math.random(),
          (Math.random() - 0.5) * 5
        );
      }
    });
  }

  stop() {
    this.leaves.forEach(l => { this.scene.remove(l); l.geometry.dispose(); l.material.dispose(); });
    this.leaves = [];
    this.active = false;
  }
}
```

## ⚠️ ÖNEMLI NOTLAR

1. **`DayNightCycle.update()`** her frame çağrılır — hafif tut, gereksiz new THREE.Color() döngü içinde oluşturma
2. **Gerçek saat kullan** — `new Date()` ile, oyun saati değil
3. **Dispose zorunlu** — sahne değişince `dayNight.dispose()` çağır
4. **Mevsim widget her dakika güncellenir** — `setInterval(renderSeasonWidget, 60000)`
5. **Hava widget her saniye güncellenir** (geri sayım için) — `setInterval(renderWeatherWidget, 1000)`
6. **Kış mevsiminde** `sunflower`, `tomato`, `blueberry` ekilemez — tohum picker'da açıkça göster
7. **Performans modunda** yıldız sayısını 200'den 50'ye düşür, yaprak sayısını 30'dan 10'a
